116 public static function get_ucd_record($code)
117 {
118 if ($code >= 0x110000) {
119 $index = 0;
120 } else {
121 $index = self::$index0[$code >> (8)] << 5;
122 $offset = ($code >> 3) & ((1 << 5) - 1);
123 $index = self::$index1[$index + $offset] << 3;
124 $offset = $code & ((1 << 3) - 1);
125 $index = self::$index2[$index + $offset];
126 }
127 return self::$ucd_records[$index];
128 }
