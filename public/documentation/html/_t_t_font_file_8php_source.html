341 function sub32($x, $y)
342 {
343 $xlo = $x[1];
344 $xhi = $x[0];
345 $ylo = $y[1];
346 $yhi = $y[0];
347
348 if ($ylo > $xlo) {
349 $xlo += 1 << 16;
350 ++$yhi;
351 }
352 $reslo = $xlo - $ylo;
353 if ($yhi > $xhi) {
354 $xhi += 1 << 16;
355 }
356 $reshi = $xhi - $yhi;
357 $reshi &= 0xFFFF;
358
359 return [$reshi, $reslo];
360 }

362 function calcChecksum($data)
363 {
364 if (strlen($data) % 4) {
365 $data .= str_repeat("\0", 4 - (strlen($data) % 4));
366 }
367
368 $len = strlen($data);
369 $hi = 0x0000;
370 $lo = 0x0000;
371
372 for ($i = 0; $i < $len; $i += 4) {
373 $hi += (ord($data[$i]) << 8) + ord($data[$i + 1]);
374 $lo += (ord($data[$i + 2]) << 8) + ord($data[$i + 3]);
375 $hi += ($lo >> 16) & 0xFFFF;
376 $lo &= 0xFFFF;
377 }
378
379 $hi &= 0xFFFF;
380
381 return [$hi, $lo];
